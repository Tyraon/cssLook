	tShadow(id[,color[,posX,posY,cou]])	//Versieht den Text des Elements(id) mit einem Textschatten.
				  "color" ist optional für die Farbe des Schattens(Format: "#000000" oder "#000" oder "black")
				  "posX", "posY" und "cou" sind optional und bestimmen die Richtung und den Verlauf
	bShadow(id[,color[,posX,posY,cou[,dis[,direction]]]])	//Versieht das Elements(id) mit einem Schatten.
				  "color" ist optional für die Farbe des Schattens(Format: "#000000" oder "#000" oder "black")
				  "posX" und "posY" geben die Richtung des Schattens an(optional) und "cou" den Härtegrad/Verlauf(optional)
				  Mit "dis" kann man die Distanz zum Element setzen(optional)
				  "direction" setzt den Schatten in das Element(optional)
	gradBg(id[,color1,color2]) //Versieht das Element mit einem Hintergrundverlauf.
				"color1" und "color2" sind optional für die Farben des Verlaufs(Format(ausschließlich): "#000000")
	boRad(id[,radius])	//Rundet die Ecken von dem Element(id) ab. "radius" ist optional und gibt den Radius der Abrundung an(Format: "5")
	zoom(id,width,height)	//Vergrößert(zoomt) das Bild(id) auf 200%. "width" und "height" sind erforderlich da sonst original größe verwendet wird
	persX(id,rotation)	//Kippt das Element(id) in dem angegebenen Winkel(rotation) um die X-Achse
	persY(id,rotation)	//Kippt das Element(id) in dem angegebenen Winkel(rotation) um die Y-Achse
	skew(id,X,Y)		//Verschiebt von dem Element(id) die Kanten um die angegebenen Winkel(X,Y) und lässt das Element rautenförmig aussehen.
	trans(id,posX,posY)	//Verschiebt die Position des Elements(id) um die angegebenen Werte(posX,posY)
	deep(id,color)		//Versieht das Element(id) mit einer räumlichen Tiefe in der angegebenen Farbe(color)
	tDeep(id,color)		//versieht den Text(id) mit einer räumlichen Tiefe in der angegebenen Farbe(color)
	puls(id,color,duration)	//Lässt das Element(id) mit der angegebenen Farbe(color) in dem Zeitabstand(duration) pulsieren
